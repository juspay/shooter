# Phase 0 — Sequencing Foundation

**Phase:** 0 of 7  
**Research anchor:** §6 "Phase 0 — Sequencing foundation" in `docs/superpowers/research/2026-06-13-terminal-sync-research.md`  
**Gaps addressed:** G3 (ordering — prevents scrollback/live race), groundwork enabling G1 (divergence) and G7 (resume)  
**Design decisions in scope:** None of D1–D5 are activated yet; this is a pure wire-protocol extension.  
**No user-visible behavior change.** All existing clients that do not send/read `seq` continue to work as before.

---

## 1. What and Why

Today every output frame broadcast by `broadcastOutput()` (pty-manager.ts:523) is a bare
`{type:'output', data}`. There is no ordering signal — a reconnecting client cannot tell
the server "I have seen up to chunk N, give me from N+1." This makes the scrollback/live
join race (G3) and silent divergence (G1) structurally unfixable: without a shared counter
there is no "from where?" to replay from.

Phase 0 adds **one monotonic integer — `seq`** — to every broadcast output frame. The server
maintains a per-terminal counter and a bounded `{seq, data}[]` ring alongside (but not
replacing) the existing raw scrollback string. The client tracks its highest-seen `seq` in
`lastSeq` but does nothing with it yet (Phase 2 will send it on reconnect). The ring is the
relay buffer Phase 2 will use for gap-fill; the counter is the ordering primitive Phase 1 will
use to attach the live tail to the snapshot without duplication.

**Invariant after Phase 0:** every `output` frame arriving at a client carries a
monotonically-increasing `seq` field; `lastSeq` on the client always equals the highest
`seq` received.

/**
 * Unit tests for the Phase 0 sequence ring logic.
 *
 * These tests replicate the exact functions that will be added to pty-manager.ts
 * (appendSeqRing, getSeqRing, getSeqRingFrom) as pure CJS functions — no TypeScript,
 * no module aliases needed.
 */
'use strict';

const assert = require('node:assert');

// ── Constants (must match pty-manager.ts after the change) ──────────────────
const SEQ_RING_MAX_ENTRIES = 2000;

// ── Pure implementations under test ────────────────────────────────────────

/**
 * Append one entry to the ring. Trims oldest when over the cap.
 * Mutates ring in place; returns the new seq (= old counter + 1).
 */
function appendSeqRing(ring, counter, data) {
  const seq = counter + 1;
  ring.push({ data, seq });
  if (ring.length > SEQ_RING_MAX_ENTRIES) {
    ring.shift();
  }
  return seq;
}

/**
 * Return all entries with seq > afterSeq, in order.
 * Returns null if afterSeq < ring[0].seq (gap too large — caller must snapshot).
 */
function getSeqRingFrom(ring, afterSeq) {
  if (ring.length === 0) return [];
  if (afterSeq < ring[0].seq - 1) return null; // gap
  return ring.filter((e) => e.seq > afterSeq);
}

// ── Tests ───────────────────────────────────────────────────────────────────

let passed = 0;
function test(name, fn) {
  fn();
  passed++;
  console.log(`  ok ${name}`);
}

console.log('seq-ring:');

test('seq starts at 1 for an empty ring', () => {
  const ring = [];
  const seq = appendSeqRing(ring, 0, 'hello');
  assert.strictEqual(seq, 1);
  assert.strictEqual(ring.length, 1);
  assert.deepStrictEqual(ring[0], { data: 'hello', seq: 1 });
});

test('seq increments monotonically', () => {
  const ring = [];
  let counter = 0;
  counter = appendSeqRing(ring, counter, 'a');
  counter = appendSeqRing(ring, counter, 'b');
  counter = appendSeqRing(ring, counter, 'c');
  assert.strictEqual(ring.map((e) => e.seq).join(','), '1,2,3');
  assert.strictEqual(counter, 3);
});

test('ring trims oldest entry when cap exceeded', () => {
  const ring = [];
  let counter = 0;
  for (let i = 0; i < SEQ_RING_MAX_ENTRIES + 1; i++) {
    counter = appendSeqRing(ring, counter, `chunk-${i}`);
  }
  assert.strictEqual(ring.length, SEQ_RING_MAX_ENTRIES);
  assert.strictEqual(ring[0].seq, 2); // seq 1 was evicted
  assert.strictEqual(ring[ring.length - 1].seq, SEQ_RING_MAX_ENTRIES + 1);
});

test('getSeqRingFrom returns entries after afterSeq in order', () => {
  const ring = [];
  let counter = 0;
  for (let i = 0; i < 5; i++) counter = appendSeqRing(ring, counter, `d${i}`);
  const tail = getSeqRingFrom(ring, 3);
  assert.deepStrictEqual(
    tail.map((e) => e.seq),
    [4, 5]
  );
});

test('getSeqRingFrom returns [] when afterSeq equals highest seq', () => {
  const ring = [];
  let counter = 0;
  for (let i = 0; i < 3; i++) counter = appendSeqRing(ring, counter, `x${i}`);
  const tail = getSeqRingFrom(ring, 3);
  assert.deepStrictEqual(tail, []);
});

test('getSeqRingFrom returns null when gap is before ring start', () => {
  const ring = [];
  let counter = 0;
  // Fill past cap so ring[0].seq > 1
  for (let i = 0; i < SEQ_RING_MAX_ENTRIES + 10; i++) {
    counter = appendSeqRing(ring, counter, 'y');
  }
  // afterSeq = 0 is before ring start (ring[0].seq = 11)
  const result = getSeqRingFrom(ring, 0);
  assert.strictEqual(result, null);
});

test('getSeqRingFrom returns all entries when afterSeq = 0 and ring fits', () => {
  const ring = [];
  let counter = 0;
  for (let i = 0; i < 5; i++) counter = appendSeqRing(ring, counter, `z${i}`);
  const tail = getSeqRingFrom(ring, 0);
  assert.strictEqual(tail.length, 5);
  assert.strictEqual(tail[0].seq, 1);
});

test('data field is preserved exactly (including VT escape sequences)', () => {
  const ring = [];
  const vt = '\x1b[?1049h\x1b[H\x1b[2J'; // enter alt screen, home, clear
  appendSeqRing(ring, 0, vt);
  assert.strictEqual(ring[0].data, vt);
});

console.log(`\n${passed} passed`);
